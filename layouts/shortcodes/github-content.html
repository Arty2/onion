{{- /* github-content — fetch a GitHub repository's README and inline its rendered Markdown. SECURITY: the fetched README is untrusted remote content. Hugo's markdownify runs shortcode rendering BEFORE markdown, so any literal shortcode delimiter in the README would be executed against the consumer site's template registry — a server-side template injection vector. We replace the opening `{{` with the fullwidth lookalike `｛｛` (U+FF5B) so the literal byte sequence `{{<` / `{{%` never appears. Fullwidth survives both plainification (no entity decode) and markdown rendering (no escape unwrap), so it stays neutralised even when downstream templates re-pass `.Summary` through markdownify. */ -}} {{- $repository := "" -}} {{- $branch := "master" -}} {{- $data := dict -}} {{- $repository := .Get "repository" -}} {{- $branch := default $branch (.Get "branch") -}} {{- $readme := printf "https://api.github.com/repos/%s/readme" $repository -}} {{- $content := printf `%s %s` (i18n "github-error") $repository $repository -}} {{- $branchURL := printf "https://raw.githubusercontent.com/%s/%s/" $repository $branch -}} {{ with resources.GetRemote $readme }} {{ with .Err }} {{ warnf "Github-Content: Error %s" . }} {{ else }} {{ $data = . | transform.Unmarshal }} {{- if gt (len $data) 0 -}} {{- $content = $data.content | base64Decode -}} {{/* remove title if same with the document’s H1 */}} {{ with $.Page.Params.Title -}} {{ $content = replace $content (printf "# %s" . ) "" }} {{ end }} {{/* make links absolute, useful for images */}} {{- $content = replace $content "](./" (printf "%s%s" "](" $branchURL) -}} {{- $content = replace $content "](/" (printf "%s%s" "](" $branchURL) -}} {{/* SECURITY: neutralise Hugo shortcode delimiters in the remote README. See top-of-file comment for rationale. */}} {{- $content = replace $content "{{<" "｛｛<" -}} {{- $content = replace $content "{{%" "｛｛%" -}} {{- end -}} {{- $content | markdownify -}} {{ end }} {{ else }} {{ warnf "Github-Content: Unable to get README from %q at %s" $repository .Page.Title }} {{ end }}
